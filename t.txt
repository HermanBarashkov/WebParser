 System.setProperty("webdriver.chrome.driver", "chromedriver-win64/chromedriver.exe");

        WebDriver driver = new ChromeDriver();
        try {
            driver.get("https://www.sberbank.com/ru/person/contributions/deposits/luchshiy_procent");

            // Находим элементы формы и вводим данные
            WebElement amountField = driver.findElement(By.name("depositAmount")); // Замените на реальное имя поля// Замените на реальный селектор
            amountField.sendKeys("100000");

            // Ждем, пока страница загрузится и результат появится
            Thread.sleep(5000); // Можно использовать более надежное ожидание

            // Парсим ответ
            WebElement resultElement = driver.findElement(By.cssSelector("div.dk-sbol-heading.dk-sbol-heading_size_lg.dk-sbol-heading_grey.dc-result-card__revenue")); // Замените на реальный селектор
            String result = resultElement.getText();

            // Сохраняем результат в Excel
            List<String[]> allData = new ArrayList<>();
            allData.add(new String[]{"Результат", result});
            writeDataToExcel(allData, "output.xlsx");
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            driver.quit();
        }

    }
